input_submit.addEventListener("click", () => {
    let isi_nama = document.getElementById("nama").value;
    let isi_kegiatan = document.getElementById("kegiatan").value;
    let isi_waktu = document.getElementById("waktu").value;
    let bagian_tombol = document.getElementsByClassName("control")[0]
    let table = document.querySelector("table");

    var old_msg = document.getElementById("error-msg");
    if(old_msg) {
        old_msg.remove()
    };

    if(isi_nama === "" || isi_kegiatan === "")  {
        var p = document.createElement("p");
        p.id = "error-msg";
        p.textContent = "tolong isi nama atau kegiatan terlebih dahulu"
        bagian_tombol.appendChild(p);
    } else {
        var tr = document.createElement("tr");
        table.appendChild(tr);
        tr.className = table.rows.length;
        tr.style.animationName = "kanankekiri";
        tr.style.animationTimingFunction = "ease";
        tr.style.animationFillMode = "forward";
        tr.style.animationDuration = "0.5s";

        var selesai = document.createElement("button");
        selesai.textContent = "✔";
        selesai.style.borderRadius = "50%";
        tr.appendChild(selesai);

        var nama = document.createElement("td");
        nama.textContent = isi_nama;
        tr.appendChild(nama);
        
        var kegiatan = document.createElement("td");
        kegiatan.textContent = isi_kegiatan;
        tr.appendChild(kegiatan);
        
        var waktu = document.createElement("td");
        waktu.textContent = isi_waktu;
        if(waktu.textContent === "" || waktu.textContent === null) {
            waktu.textContent = "-";
        };
        tr.appendChild(waktu);

        selesai.addEventListener("click", () => {
            tr.remove()
            removeTodo()
        });

    }
});
